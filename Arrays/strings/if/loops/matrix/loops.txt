import java.util.Scanner;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        
        for (int i = 0; i < T; i++) {
            int N = sc.nextInt();
            int negativeCount = 0;
            int zeroCount = 0;
            
            for (int j = 0; j < N; j++) {
                int num = sc.nextInt();
                if (num < 0) {
                    negativeCount++;
                } else if (num == 0) {
                    zeroCount++;
                }
            }
       
            if (zeroCount > 0) {
                System.out.println(0);
            } else if (negativeCount % 2 == 1) {
                System.out.println(1);
            } else {
                System.out.println(0);
            }
        }
        
        sc.close(); 
    }
}

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        int T = sc.nextInt();  
        
        while (T-- > 0) {
            int N = sc.nextInt();  
            
            int totalSum = 0;
            int minValue = Integer.MAX_VALUE;
            
            for (int i = 0; i < N; i++) {
                int value = sc.nextInt();
                totalSum += value;
                
                if (value < minValue) {
                    minValue = value;
                }
            }
            
            System.out.println(totalSum - minValue);
        }
        
        sc.close();
    }
}
