Source code:

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();

        if (a < b) {
            System.out.println("a < b");
        } else {
            if (a > b) {
                System.out.println("a > b");

public class Main {
    public static void main(String[] args) {

        int n= sc.nextInt();
